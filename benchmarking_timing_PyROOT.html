def benchmark(series):
    filepath=[]
    start = time.time()
    filepath = filepath+glob.glob(series+"/umn_*.root")
    eT = ROOT.TChain("rqDir/eventTree")
    zip1 = ROOT.TChain("rqDir/zip1")
    for file in filepath:
       eT.Add(file)
       zip1.Add(file)
    zip1.AddFriend(eT)
    print(zip1.GetEntries())
    zip1.Draw("EventTime:PTOFamps")
    end = time.time()
    print(end - start)
    c.cd(2)
    zip1.Draw("PTOFamps")
    print(time.time() - start)
    print("(",time.time()-end,")")
